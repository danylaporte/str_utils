Expand description

Trait for equality comparisons of string and chars.

This trait allow to do accent and case insensitive comparisons.

Required methods

fn eq_ai(self, other: Rhs) -> bool where
    Self: Sized, [src]

Accent insensitive

Example

use str_utils::cmp::EqExt;

// string comparison
assert!("Café".eq_ai("Cafe"));

// char comparison
assert!('e'.eq_ai('é'));

Accent / Case insensitive

Example

use str_utils::cmp::EqExt;

// string comparison
assert!("Café".eq_ai_ci("cafe"));

// char comparison
assert!('e'.eq_ai_ci('É'));

fn eq_ci(self, other: Rhs) -> bool where
    Self: Sized, [src]

Case insensitive

Example

use str_utils::cmp::EqExt;

// string comparison
assert!("street".eq_ci("Street"));

// char comparison
assert!('C'.eq_ci('c'));

// owned string comparison
assert!("abc".to_owned().eq_ci("ABC"));
